# 39: Språkkriget når sin kulmen - vilken sida väljer du?
so_reunion_fin.39 = {
	type = country_event
	
	event_image = {
		video = "gfx/event_pictures/europenorthamerica_springtime_of_nations.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/springtime_of_nation"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	title = so_reunion_fin.39.t
	desc = {
		first_valid = {
			triggered_desc = {
				desc = so_reunion_fin.39.d_1
				trigger = {
					AND = {
						var:so_var_fin_svecoman_progress > var:so_var_fin_fennoman_progress
						var:so_var_fin_svecoman_progress > var:so_var_fin_russoman_progress
					}
				}
			}
			triggered_desc = {
				desc = so_reunion_fin.39.d_2
				trigger = {
					AND = {
						var:so_var_fin_fennoman_progress > var:so_var_fin_svecoman_progress
						var:so_var_fin_fennoman_progress > var:so_var_fin_russoman_progress
					}
				}
			}
			triggered_desc = {
				desc = so_reunion_fin.39.d_3
				trigger = {
					AND = {
						var:so_var_fin_russoman_progress > var:so_var_fin_fennoman_progress
						var:so_var_fin_russoman_progress > var:so_var_fin_svecoman_progress
					}
				}
			}
		}
	}
	flavor = {
		first_valid = {
			triggered_desc = {
				desc = so_reunion_fin.39.f_1
				trigger = {
					AND = {
						var:so_var_fin_svecoman_progress > var:so_var_fin_fennoman_progress
						var:so_var_fin_svecoman_progress > var:so_var_fin_russoman_progress
					}
				}
			}
			triggered_desc = {
				desc = so_reunion_fin.39.f_2
				trigger = {
					AND = {
						var:so_var_fin_fennoman_progress > var:so_var_fin_svecoman_progress
						var:so_var_fin_fennoman_progress > var:so_var_fin_russoman_progress
					}
				}
			}
			triggered_desc = {
				desc = so_reunion_fin.39.f_3
				trigger = {
					AND = {
						var:so_var_fin_russoman_progress > var:so_var_fin_fennoman_progress
						var:so_var_fin_russoman_progress > var:so_var_fin_svecoman_progress
					}
				}
			}
		}
	}
	
	duration = 3

	cooldown = { years = 10 }
	
	# triggered when either group reaches the goal of 200
	trigger = {
		OR = {
			var:so_var_fin_svecoman_progress >= 200
			var:so_var_fin_fennoman_progress >= 200
			var:so_var_fin_russoman_progress >= 200
		}
	}

	option = { # Svekomanerna har min röst
		name = so_reunion_fin.39.a
        default_option = yes
		if = {
			limit = {
				var:so_var_fin_svecoman_progress >= 200
			}
			highlighted_option = yes
		}
		set_variable = { name = so_var_fin_vtf_main_complete }
		add_journal_entry = { type = so_var_fin_vtf_end }
		set_variable = { name = so_var_fin_vtf_end_svecoman }
		ai_chance = {
			base = 40
		}
	}
	
	option = { # Framtiden finns hos fennomanerna
		name = so_reunion_fin.39.b
		if = {
			limit = {
				var:so_var_fin_fennoman_progress >= 200
			}
			highlighted_option = yes
		}
		set_variable = { name = so_var_fin_vtf_main_complete }
		add_journal_entry = { type = so_var_fin_vtf_end }
		set_variable = { name = so_var_fin_vtf_end_fennoman }
		ai_chance = {
			base = 40
		}
	}
	
	option = { # All makt åt Tsaren, vår befriare
		name = so_reunion_fin.39.c
		if = {
			limit = {
				var:so_var_fin_russoman_progress >= 200
			}
			highlighted_option = yes
		}
		set_variable = { name = so_var_fin_vtf_main_complete }
		add_journal_entry = { type = so_var_fin_vtf_end }
		set_variable = { name = so_var_fin_vtf_end_russoman }
		ai_chance = {
			base = 20
		}
	}
}

# 41: Forced conscription
so_reunion_fin.41 = {
	type = country_event
	placement = ROOT

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	event_image = {
		video = "votp_garibaldi"
	}
	
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	title = so_reunion_fin.41.t
	desc = so_reunion_fin.41.d
	flavor = so_reunion_fin.41.f
	
	duration = 3

	cooldown = { years = 3 }
	
	# triggered by VtF_1
	trigger = {
	}
	
	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				has_building = building_conscription_center
			}
			save_scope_as = ev41_state_scope
		}
		if = {
			limit = {
				scope:ev41_state_scope = {
					any_scope_building = {
						is_building_type = building_conscription_center
					}
				}
			}
			scope:ev41_state_scope = {
				random_scope_building = {
					limit = {
						is_building_type = building_conscription_center
					}
					save_scope_as = ev41_building_scope
				}
			}
		}
		c:RUS = {
			random_scope_state = {
				limit = {
					is_incorporated = yes
					has_building = building_conscription_center
				}
				save_scope_as = ev41_rus_state_scope
			}
		}
	}

	option = { # What if we all hide in the woods?
		name = so_reunion_fin.41.a
        default_option = yes
		if = {
			limit = {
				c:RUS = {
					relations:root < relations_threshold:poor
				}
			}
			change_relations = {
				country = c:RUS
				value = -10
			}
		}
		else = {
			change_relations = {
				country = c:RUS
				value = -5
			}
		}
		scope:ev41_building_scope = {
			add_modifier = {
				name = so_fin_mod_ev41_passive_resistance_building
				years = 1
				multiplier = 1
			}
		}
		scope:ev41_state_scope = {
			add_modifier = {
				name = so_fin_mod_ev41_passive_resistance
				years = 1
				multiplier = 1
			}
		}
		ai_chance = {
			base = 50
		}
	}
	
	option = { # Our patience is running out. Take an active stance against the recruitment officer!
		name = so_reunion_fin.41.b
		if = {
			limit = {
				c:RUS = {
					relations:root < relations_threshold:poor
				}
			}
			change_relations = {
				country = c:RUS
				value = -20
			}
		}
		else = {
			change_relations = {
				country = c:RUS
				value = -10
			}
		}
		scope:ev41_building_scope = {
			add_modifier = {
				name = so_fin_mod_ev41_active_resistance_building
				years = 1
				multiplier = 1
			}
		}
		scope:ev41_state_scope = {
			add_modifier = {
				name = so_fin_mod_ev41_active_resistance
				years = 1
				multiplier = 1
			}
		}
	ai_chance = {
			base = 30
		}
	}
	
	option = { # The Tsar, our dear leader, would want nothing else than a few able-bodied men. Is that so much to ask for?
		name = so_reunion_fin.41.c
		if = {
			limit = {
				c:RUS = {
					relations:root < relations_threshold:poor
				}
			}
			change_relations = {
				country = c:RUS
				value = 5
			}
		}
		else = {
			change_relations = {
				country = c:RUS
				value = 10
			}
		}
		scope:ev41_state_scope = {
			add_modifier = {
				name = so_fin_mod_ev41_no_resistance
				years = 1
				multiplier = 1
			}
		}
		scope:ev41_rus_state_scope = {
			c:RUS = {
				add_modifier = {
					name = so_fin_mod_ev41_no_resistance_rus
					years = 1
					multiplier = 1
				}
			}
		}
		ai_chance = {
			base = 50
		}
	}
}
