        default_option = yes

			limit = {
				is_set = root.var:so_var_fin_so_vtf_1

			change_variable = {
				name = so_var_fin_so_vtf_1
				add = 1
			}
		}
		else = {
			set_variable = {
				name = so_var_fin_so_vtf_1
				value = 1
			}
		}

			country = c:RUS
			value = -5
		}

				has_modifier = so_fin_mod_ev41_passive_resistance
