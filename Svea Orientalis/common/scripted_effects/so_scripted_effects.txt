so_set_fin_russification_variables = {
	#Constitutionalist Variables
	set_variable = { name = so_constitutionalist_progress_from_movements value = 0 } #Progress-from-IG component
	set_variable = { name = so_constitutionalist_progress_from_effects value = 0 } #This is defined by various effects, and is based solely on scripted effects. This is how designers can manually grant ticking bonuses or maluses to this mechanic.
	
	#Appeasement Variables
	set_variable = { name = so_appeasement_progress_from_movements value = 0 } #Progress-from-IG component
	set_variable = { name = so_appeasement_progress_from_effects value = 0 } #This is defined by various effects, and is based solely on scripted effects. This is how designers can manually grant ticking bonuses or maluses to this mechanic.
	
	#Activist Variables
	set_variable = { name = so_activist_progress_from_movements value = 0 } #Progress-from-IG component
	set_variable = { name = so_activist_progress_from_effects value = 0 } #This is defined by various effects, and is based solely on scripted effects. This is how designers can manually grant ticking bonuses or maluses to this mechanic.
}

so_weekly_pulse_constitutionist_progress = { #Effects like these are run three times, one for each faction.
	set_variable = { name = so_constitutionalist_progress_from_movements value = 0 }
	# From Movements
	if = {
		limit = {
			any_political_movement = {
				is_political_movement_type = movement_constitutionalist
			}
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_constitutionalist
			}
			ROOT = {
				set_variable = { name = so_constitutionalist_progress_from_movements value = prev.political_movement_support }
				change_variable = { name = so_constitutionalist_progress_from_movements multiply = 10 }
			}
		}
	}
}
