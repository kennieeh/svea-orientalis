so_je_swe_so_1 = { # It is Gone, but not Forgotten
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	is_shown_when_inactive = {
		exists = c:SWE
		this = c:SWE
		c:FIN = {
			is_subject = yes
			NOT = { is_subject_of = ROOT }
		}
	}

	# 
	possible = {
		has_technology_researched = nationalism
	}
	
	# Every month, there is a chance for an event to happen (but not a big chance - 20/100) - each event increases progress bar by one. TODO: More events!
	on_monthly_pulse = {
		random_events = {
			10 = so_reunion_swe.11
			10 = so_reunion_swe.12
			80 = 0
		}
	}
	
	# 
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	
	# När avklarat, trigga eventet där man väljer om man ska fokusera på FIN eller inte
	on_complete = {
		trigger_event = so_reunion_swe.10
	}

	current_value = {
		value = root.var:so_var_swe_so_1
	}

	goal_add_value = {
		add = 10
	}

	progressbar = yes
	
	# Timeout är 100 år... Så att den inte timeoutar :)
	timeout = 36500
	should_be_pinned_by_default = yes
}

so_je_fin_vtf_start = { # Start of the journal entries
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	is_shown_when_inactive = {
		exists = c:FIN
		this = c:FIN
		c:FIN = {
			is_subject_of = c:RUS
		}
	}
	
	
	possible = {
		always = yes
	}
	
	complete = {
		has_technology_researched = nationalism
	}

	on_complete = {
		add_journal_entry = { type = so_je_fin_vtf_1 }
		set_variable =  {
			name = so_var_fin_so_vtf_1
			value = 0
		}
	}
}

so_je_fin_vtf_1 = { # Resist the Czar
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	# triggered by so_je_fin_vtf_start

	# Varje år händer ett av dessa event (eller 0, dvs ingenting), angivet som procent av 100
	on_yearly_pulse = {
		random_events = {
			80 =  0
			5 = so_reunion_fin.41
			5 = so_reunion_fin.42
			5 = so_reunion_fin.43
			5 = so_reunion_fin.44
			5 = so_reunion_fin.45
			5 = so_reunion_fin.46
			5 = so_reunion_fin.47
			5 = so_reunion_fin.48
			5 = so_reunion_fin.49
		}
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = so_reunion_fin.40
	}

	current_value = {
		value = root.var:so_var_fin_so_vtf_1
	}

	# Håller på tills goal är klart, goal har värdet 10, varje event som händer lägger till 1 = 10 events
	goal_add_value = {
		add = 10
	}

	progressbar = yes
	
	should_be_pinned_by_default = yes
}
